StackLayoutView¶

Dieses layout wird in main.qml im unteren Bereich als Tabulartor- oder Registeransicht verwendet. Der Benutzer kann auf Knopfdruck zwischen den drei Layouts CommissionView, EventView oder Inventoryview.qml wechseln. Das Layout is in der Datei StackLayoutView.qml implementiert:

 1import QtQuick 2.9
 2import QtQuick.Controls 2.15
 3import QtQuick.Controls.Material
 4import QtQuick.Layouts 1.15
 5
 6Rectangle {
 7    Material.accent: Material.DarkBlue
 8    Material.primary: Material.Dark
 9    Material.foreground: Material.White
10    Material.background: Material.Black
11    TabBar {
12        id: tabBar
13        width: parent.width
14        height: 50
15        TabButton {
16            text: "Commission"
17            onClicked: stackLayout.currentIndex = 0
18        }
19
20        TabButton {
21            text: "Eventlog"
22            onClicked: stackLayout.currentIndex = 1
23        }
24
25        TabButton {
26            text: "Inventory"
27            onClicked: stackLayout.currentIndex = 2
28        }
29    }
30    StackLayout {
31        id: stackLayout
32        anchors.top: tabBar.bottom
33        anchors.left: parent.left
34        anchors.right: parent.right
35        anchors.bottom: parent.bottom
36        currentIndex: 0
37        CommissionView {
38            id: commissionView
39            width: parent.width
40            height: parent.height
41        }
42        EventView {
43            id: eventView
44            width: parent.width

46        }

48            id: inventoryView
49            width: parent.width
50            height: parent.height
51        }
52
53    }
54}
