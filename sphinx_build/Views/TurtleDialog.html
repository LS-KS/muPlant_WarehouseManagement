Dialog: Mobilen Roboter überschreiben¶

Dieser Dialog dient dazu den Inhalt des mobilen Roboters manuell zu überschreiben. Die Implementierung erfolgt in der Datei TurtleDialog.qml:

 1import QtQuick 2.15
 2import QtQuick.Dialogs
 3import QtQuick.Controls 2.15
 4import QtQuick.Controls.Material 2.15
 5import QtQuick.Layouts 1.3
 6/*
 7  This QML File shows a Dialog which enables the user to manually override the workbench data
 8  */
 9Dialog {
10    id: turteDialog
11    title: "Override mobile Robot"
12    width: 350
13    // ColumnLayout helps to organize Items in vertical order.
14    ColumnLayout{
15        anchors.fill: parent
16        Layout.fillHeight: true
17        Layout.fillWidth: true
18
19        // This row has a textlabel and textfield which enables the user to override Cup ID
20        Row{
21            Text {
22                id: cupText
23                width: parent.width/2
24                height: setCup.height
25                text: qsTr("Set Cup ID: ")
26                verticalAlignment: Text.AlignVCenter
27                Layout.fillHeight: true
28                Layout.fillWidth: true
29            }
30            TextField{
31                id: setCup
32                // limit the cup ID to positive integer between 0 and 9999
33                validator: IntValidator{
34                    bottom: 0
35                    top: 9999
36                }
37            }
38            Layout.fillHeight: true
39            Layout.fillWidth: true
40        }
41        // This row enables the user to override product id in mobile robot
42        Row{
43            Text {
44                id: setProd
45                width: parent.width/2
46                height: setProduct.height
47                text: qsTr("Set Product ID:")
48                Layout.fillHeight: true
49                Layout.fillWidth: true
50                verticalAlignment: Text.AlignVCenter
51            }
52            ComboBox{
53                id:setProduct
54                model: productListModel
55                textRole: 'name'
56                valueRole: 'id'
57            }
58            Layout.fillHeight: true
59            Layout.fillWidth: true
60        }
61        // clearbutton enables the user to set values for cup and product which implicate that the storage is empty
62        DialogButtonBox{
63            Button {
64                id: clearButton
65                text: "Clear"
66                onClicked: {
67                    console.log("Clear in WorkbenchDialog clicked")
68                    setProduct.currentIndex = 0
69                    setCup.text = "0"
70                }
71            }
72
73        }
74    }
75
76    // standardbuttons are buttons which perform standard tasks.
77    standardButtons: Dialog.Ok | Dialog.Cancel
78    // signal which is emitted when Dialog.OK is clicked. It calls changeStorage() function of InventoryController
79    onAccepted: {
80        console.log("Change sent to InventoryController from TurtleDialog")
81        console.log("cup: " + setCup.text)
82        console.log("product: " + setProduct.currentValue)
83        inventoryController.changeMobileRobot(setCup.text, setProduct.currentValue)
84        console.log("Ok clicked in TurtleDialog")
85    }
86    onRejected: console.log("Cancel in turtleDialog clicked")
87}
