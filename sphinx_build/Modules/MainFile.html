Main File¶

Das main-File ist der Einsprungspunkt des Programms. Es werden eine Instanz der Application und der QML-Engine erstellt. Sämtliche Controller und Datenmodelle werden initialisiert. Diese Instanzen der Python-Objekte werden als RootContext der QML-Engine hinzugefügt, sodass Sie über den QML-Inhalt angesprochen werden können.

 2
 3'''
 4µPlant WareHouse Management Software.
 5Bachelor Thesis 2023
 6Author: L.Schink
 7Date Project Start: 20.06.2023
 8Language : Python 3.11
 9
10This Software implements warehouse logic from heterogene modular autonomic production site.
11The warehouse works autonomous while receiving orders from modbus TCP/IP.
12also OPC UA is implemented.
13
14Uses  PySide6 with PyQt6.5.1 for GUI implementation
15'''
16import sys
17from pathlib import Path
18from PySide6.QtGui import QGuiApplication
19from PySide6.QtQml import QQmlApplicationEngine, qmlRegisterType
20from PySide6.QtCore import Qt
21from src.controller.invController import invController
22from src.controller.CommissionController import CommissionController
23from src.controller.PreferenceController import PreferenceController
24from src.controller.RfidController import RfidController
25from src.service.EventlogService import EventlogService
26from src.service.OpcuaService import OpcuaService
27from src.constants.Constants import Constants
28from src.service.AgentService import AgentService
29
30
31if __name__ == '__main__':
32    app = QGuiApplication(sys.argv)
33    engine = QQmlApplicationEngine()
34    constants = Constants()
35    # create inventoryController with included Data Model and sets itself and viewModels  as rootContext
36    inventoryController = invController()
37
38    engine.rootContext().setContextProperty("inventoryController", inventoryController)
39    engine.rootContext().setContextProperty('storageModel', inventoryController.storageViewModel)
40    engine.rootContext().setContextProperty('productListModel', inventoryController.productlistViewModel)
41    engine.rootContext().setContextProperty('productSummaryModel', inventoryController.productSummaryViewModel)
42
43    # creates EventlogService object and sets itself as rootContext
44    eventlogService = EventlogService()
45    engine.rootContext().setContextProperty("eventModel", eventlogService.eventViewModel)
46    inventoryController.eventlogService = eventlogService
47    engine.rootContext().setContextProperty("eventLogController", eventlogService)
48
49    # creates CommissionController object and sets itself as rootContext
50    commissionController =  CommissionController(inventoryController, eventlogService)
51    commissionController.commissionFilterProxyModel.autoAcceptChildRows()
52    engine.rootContext().setContextProperty("commissionController", commissionController)
53    # engine.rootContext().setContextProperty('commissionModel', commissionController.commissionFilterProxyModel)
54    engine.rootContext().setContextProperty('commissionModel', commissionController.commissionViewModel)
55
56    # creates preferenceController object and sets itself as rootContext
57    preferenceController = PreferenceController(eventlogService)
58    engine.rootContext().setContextProperty("preferenceController", preferenceController)
59
60    # creates AgentService object and sets itself as rootContext
61    agentservice = AgentService(eventlogService, preferenceController)
62    engine.rootContext().setContextProperty("agentService", agentservice)
63
64    # creates Controller and models for RFID Server Plugin
65    rfidController = RfidController()
66    engine.rootContext().setContextProperty("rfidController", rfidController)
67    engine.rootContext().setContextProperty("rfidModel", rfidController.rfidViewModel)
68    #engine.rootContext().setContextProperty("rfidModel", rfidController.rfidProxyViewModel)
69
70    # creates opcua service
71    opcuaService = OpcuaService(eventlogService, preferenceController)
72    engine.rootContext().setContextProperty("opcuaService", opcuaService)
73
74    # define load main.qml file to start application
75    qml_file =  str(Path(__file__).resolve().parent / "src" / "view" / "main.qml")
76    engine.load(qml_file)
77
78    if not engine.rootObjects():
79        sys.exit(-1)
80
81    sys.exit(app.exec())
