15import sys
16from pathlib import Path
17from PySide6.QtGui import QGuiApplication
18from PySide6.QtQml import QQmlApplicationEngine, qmlRegisterType
19from src.controller.invController import invController
20from src.service.EventlogService import EventlogService
21
22if __name__ == '__main__':
23    app = QGuiApplication(sys.argv)
24    engine = QQmlApplicationEngine()
25
26    # create inventoryController with included Data Model and sets itself and viewModels  as rootContext
27    inventoryController = invController()
28
29    engine.rootContext().setContextProperty("inventoryController", inventoryController)
30    engine.rootContext().setContextProperty('storageModel', inventoryController.storageViewModel)
31    engine.rootContext().setContextProperty('productListModel', inventoryController.productlistViewModel)
32    engine.rootContext().setContextProperty('productSummaryModel', inventoryController.productSummaryViewModel)
33
34    # creates EventlogService object and sets itself as rootContext
35    eventlogService = EventlogService()
36    inventoryController.eventlogService = eventlogService
37    engine.rootContext().setContextProperty("eventLogController", eventlogService)
38
39    # define load main.qml file to start application
40    qml_file = Path(__file__).resolve().parent / "src" / "view" / "main.qml"
41    engine.load(qml_file)
42
43    if not engine.rootObjects():
44        sys.exit(-1)
45
46    sys.exit(app.exec())
